#include <stdio.h>

int main() {
 int n, i, j, count, maxcount = 0, mostfreq;
 int arr1[100];
 printf("Enter size of array: ");
 scanf("%d", &n);
 printf("Enter %d elements:\n", n);
 for (i = 0; i < n; i++) {
 scanf("%d", &arr1[i]); 
}
// find frequency of each element
 for (i = 0; i < n; i++) {
     count = 0;
     for (j = 0; j < n; j++) {
     if (arr1[i] == arr1[j]) {
         count++;
     }
 }
 if(count > maxcount) {
     maxcount = count;
     mostfreq = arr1[i];
 }
 }
 printf("Most frequent element is %d (appears %d times)\n", mostfreq,maxcount);
    return 0;
}
