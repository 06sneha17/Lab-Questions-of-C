#include <stdio.h>
int main() {
int num1, num2;
float result;
// prompt the user to enter two integers
printf("Enter two  integers: ");
scanf("%d %d", &num1, &num2);
// perform division with type conversion to float
// casting one of the operands to flaot ensures floating-point division
result = (float)num1 / num2;
// print the result 
printf("Result: %.1f\n", result);
    return 0;
}
